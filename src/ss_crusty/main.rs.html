1//! ss_crusty (Simple Scraper) is a high-performance, command-line application written in **Rust** ðŸ¦€ designed to **fetch HTML content from a list of URLs in parallel and extract all absolute links**.
2//!
3//! ### Core Functionality
4//!
5//! 1.  **Input Versatility:** Accepts target URLs both directly via command-line arguments and from external files.
6//! 2.  **Concurrency (Multi-threading):** Utilizes the **[`rayon`](https://docs.rs/rayon/latest/rayon/)** library for **parallel execution** (`par_iter().map(...)`), enabling the application to fetch and scrape multiple URLs concurrently across available CPU cores, ensuring fast and efficient processing of large lists.
7//! 3.  **Data Extraction:** Employs the [`reqwest`](https://docs.rs/reqwest/latest/reqwest/) HTTP client and the [`scraper`](https://docs.rs/scraper/latest/scraper/) parsing library to robustly fetch and analyze web pages.
8//! 4.  **Structured Output:** Gathers all results (original URL, raw HTML, and unique extracted links) and serializes them into a single, clean **JSON file** using [`serde`](https://docs.rs/serde/latest/serde/).
9//!
10//! ---
11//!
12//! #### Arguments
13//!
14//! * **`--urls <URL>`**: URLs to fetch. **Must be repeated** for each URL to be added (e.g., `--urls "url1" --urls "url2"`).
15//! * **`--file <PATH>`**: Path to one or more files containing URLs. **Must be repeated** for each file (e.g., `--file "list1.txt" --file "list2.txt"`).
16//! * **`--output <PATH>`**: The output JSON file name (defaults to `results.json`).
17//! * **`--user-agent <AGENT:String>`**: Optional user-agent string override (`Mozilla`, `Webkit`, or `Chrome`).
18//! * **`-v, -vv, -vvv`**: Controls logging level (`-v` = Info, `-vv` = Debug, `-vvv` = Warn).
19//!
20//! ---
21//!
22//! #### Usage
23//!
24//! Run the application with URLs and specify the output file:
25//!
26//! ```bash
27//! ss_crusty --urls "https://example.com" --file "list.txt" --output "results.json" -vv
28//! ```

30mod scrape;
31mod io;

33use cli::parse_args_and_client;
34use rayon::prelude::*;
35use serde::Serialize;
36use log::{info, debug};

38#[derive(Serialize)]
39/// Represents the result of scraping a single URL.
40/// This structure is used by the application to serialize the final data into the output JSON file.
41struct PageResult {
42    /// The URL that was fetched.
43    url: String,
44    /// A list of all unique, absolute links extracted from the page's HTML.
45    links: Vec<String>,
46    /// The raw HTML content of the page.
47    html: String,
48}
49
50fn main() -> Result<(), Box<dyn std::error::Error>> {
51    let (args, client) = parse_args_and_client()?;
52
53    let log_level = match args.verbose {
54        0 => "warn",
55        1 => "info",
56        2 => "debug",
57        _ => "trace",
58    };
59    
60    unsafe { std::env::set_var("RUST_LOG", format!("ss_crusty={}", log_level)); }
61    env_logger::init();
62
63    info!("Starting ss_crusty...");
64    debug!("Arguments: {:?}", args);
65
66    if args.urls.is_empty() && args.file.is_empty() {
67        println!("No URLs or files provided.");
68        println!("Verbose usage: -v (info), -vv (debug), -vvv (trace)");
69        std::process::exit(1);
70    }
71
72    let mut urls = args.urls.clone();
73
74    // Parse all input files
75    for file_path in &args.file {
76        info!("Reading file: {}", file_path);
77        let content = io::read_file(file_path)?;
78        let parsed = parsers::parse_file_content(file_path, &content);
79        urls.extend(parsed);
80    }
81
82    urls.sort();
83    urls.dedup();
84
85    info!("Fetching {} unique URLs...", urls.len());
86
87    let results: Vec<PageResult> = urls
88        .par_iter()
89        .map(|url| {
90            let (html, links) = scrape::fetch_and_extract(&client, url);
91            PageResult {
92                url: url.to_string(),
93                links,
94                html,
95            }
96        })
97        .collect();
98
99    let wrapped = serde_json::json!({ "pages": results });
100    io::write_file(&args.output, &serde_json::to_string_pretty(&wrapped)?)?;
101    info!("Results written to {}", args.output);
102    info!("Processed {} input files", args.file.len());
103
104    Ok(())
105}
