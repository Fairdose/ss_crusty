1//! ss_crusty (Simple Scraper) is a high-performance, command-line application written in **Rust** ðŸ¦€ designed to **fetch HTML content from a list of URLs in parallel and extract all absolute links**.
2//! [GitHub Repo](https://github.com/Fairdose/ss_crusty)
3//!
4//! ### Core Functionality
5//!
6//! 1.  **Input Versatility:** Accepts target URLs both directly via command-line arguments and from external files.
7//! 2.  **Concurrency (Multi-threading):** Utilizes the **[`rayon`](https://docs.rs/rayon/latest/rayon/)** library for **parallel execution** (`par_iter().map(...)`), enabling the application to fetch and scrape multiple URLs concurrently across available CPU cores, ensuring fast and efficient processing of large lists.
8//! 3.  **Data Extraction:** Employs the [`reqwest`](https://docs.rs/reqwest/latest/reqwest/) HTTP client and the [`scraper`](https://docs.rs/scraper/latest/scraper/) parsing library to robustly fetch and analyze web pages.
9//! 4.  **Structured Output:** Gathers all results (original URL, raw HTML, and unique extracted links) and serializes them into a single, clean **JSON file** using [`serde`](https://docs.rs/serde/latest/serde/).
10//!
11//! ---
12//!
13//! #### Arguments
14//!
15//! * **`--urls <URL>`**: URLs to fetch. **Must be repeated** for each URL to be added (e.g., `--urls "url1" --urls "url2"`).
16//! * **`--file <PATH>`**: Path to one or more files containing URLs. **Must be repeated** for each file (e.g., `--file "list1.txt" --file "list2.txt"`).
17//! * **`--output <PATH>`**: The output JSON file name (defaults to `results.json`).
18//! * **`--user-agent <AGENT:String>`**: Optional user-agent string override (`Mozilla`, `Webkit`, or `Chrome`).
19//! * **`-v, -vv, -vvv`**: Controls logging level (`-v` = Info, `-vv` = Debug, `-vvv` = Warn).
20//!
21//! ---
22//!
23//! #### Usage
24//!
25//! Run the application with URLs and specify the output file:
26//!
27//! ```bash
28//! ss_crusty --urls "https://example.com" --file "list.txt" --output "results.json" -vv
29//! ```
30mod cli;
31mod scrape;
32mod io;
33
34use cli::parse_args_and_client;
35use rayon::prelude::*;
36use serde::Serialize;
37use log::{info, debug};
38
39#[derive(Serialize)]
40/// Represents the result of scraping a single URL.
41/// This structure is used by the application to serialize the final data into the output JSON file.
42struct PageResult {
43    /// The URL that was fetched.
44    url: String,
45    /// A list of all unique, absolute links extracted from the page's HTML.
46    links: Vec<String>,
47    /// The raw HTML content of the page.
48    html: String,
49}
50
51fn main() -> Result<(), Box<dyn std::error::Error>> {
52    let (args, client) = parse_args_and_client()?;
53
54    let log_level = match args.verbose {
55        0 => "warn",
56        1 => "info",
57        2 => "debug",
58        _ => "trace",
59    };
60    
61    unsafe { std::env::set_var("RUST_LOG", format!("ss_crusty={}", log_level)); }
62    env_logger::init();
63
64    info!("Starting ss_crusty...");
65    debug!("Arguments: {:?}", args);
66
67    if args.urls.is_empty() && args.file.is_empty() {
68        println!("No URLs or files provided.");
69        println!("Verbose usage: -v (info), -vv (debug), -vvv (trace)");
70        std::process::exit(1);
71    }
72
73    let mut urls = args.urls.clone();
74
75    // Parse all input files
76    for file_path in &args.file {
77        info!("Reading file: {}", file_path);
78        let content = io::read_file(file_path)?;
79        let parsed = parsers::parse_file_content(file_path, &content);
80        urls.extend(parsed);
81    }
82
83    urls.sort();
84    urls.dedup();
85
86    info!("Fetching {} unique URLs...", urls.len());
87
88    let results: Vec<PageResult> = urls
89        .par_iter()
90        .map(|url| {
91            let (html, links) = scrape::fetch_and_extract(&client, url);
92            PageResult {
93                url: url.to_string(),
94                links,
95                html,
96            }
97        })
98        .collect();
99
100    let wrapped = serde_json::json!({ "pages": results });
101    io::write_file(&args.output, &serde_json::to_string_pretty(&wrapped)?)?;
102    info!("Results written to {}", args.output);
103    info!("Processed {} input files", args.file.len());
104
105    Ok(())
106}
